  },
  {
    text: "How do you prefer to spend a weekend?",

  {
    text: "What is your favorite activity?",

    options: ["Rock", "Classical", "Jazz", "Pop"],
    correct: "Rock",
  },
  {
    text: "What is your favorite type of weather?",
    options: ["Sunny", "Rainy", "Windy", "Snowy"],
    correct: "Sunny",
  },
];

export function Quiz() {
  const [current, setCurrent] = useState(0);
  const [answers, setAnswers] = useState<string[]>([]);
  const [showResult, setShowResult] = useState(false);

  const handleAnswer = (answer: string) => {
    setAnswers([...answers, answer]);
    if (current + 1 < questions.length) {
      setCurrent(current + 1);
    } else {
      setShowResult(true);
    }
  };

  const reset = () => {
    setCurrent(0);
    setAnswers([]);
    setShowResult(false);
  };

  if (showResult) {
    return <QuizResult answers={answers} onRetake={reset} />;
  }

  const q = questions[current];
  const shuffled = shuffle([...q.options]);

  return (
    <div className="max-w-md mx-auto p-4 space-y-4">
      <h2 className="text-xl font-semibold">{q.text}</h2>
      <div className="grid gap-2">
        {shuffled.map((opt) => (
          <button
            key={opt}
            className="px-4 py-2 bg-primary text-primary-foreground rounded hover:bg-primary/80"
            onClick={() => handleAnswer(opt)}
          >
            {opt}
          </button>
        ))}
      </div>
    </div>
  );
}

"use client";

import { useEffect, useState } from "react";
import { Share } from "@/components/share";
import { url } from "@/lib/metadata";

type Props = {
  answers: string[];
  onRetake: () => void;
};

const animalScores = {
  cat: 0,
  dog: 0,
  fox: 0,
  hamster: 0,
  horse: 0,
};

const animalImages = {
  cat: "/cat.png",
  dog: "/dog.png",
  fox: "/fox.png",
  hamster: "/hamster.png",
  horse: "/horse.png",
};

export function QuizResult({ answers, onRetake }: Props) {
  const [result, setResult] = useState<string>("");

  useEffect(() => {
    const scores = { ...animalScores };
    answers.forEach((ans) => {
      if (ans === "Meat") scores.dog += 1;
      if (ans === "Vegetables") scores.hamster += 1;
      if (ans === "Fish") scores.cat += 1;
      if (ans === "Sweet treats") scores.fox += 1;
      if (ans === "Sleeping") scores.cat += 1;
      if (ans === "Running around") scores.horse += 1;
      if (ans === "Exploring") scores.horse += 1;
      if (ans === "Chasing") scores.dog += 1;
      if (ans === "Sniffing") scores.dog += 1;
      if (ans === "Hiding") scores.hamster += 1;
      if (ans === "Running") scores.horse += 1;
      if (ans === "Rock") scores.dog += 1;
      if (ans === "Classical") scores.cat += 1;
      if (ans === "Jazz") scores.fox += 1;
      if (ans === "Pop") scores.dog += 1;
      if (ans === "Sunny") scores.horse += 1;
      if (ans === "Rainy") scores.cat += 1;
      if (ans === "Windy") scores.horse += 1;
      if (ans === "Snowy") scores.fox += 1;
    });

    const best = Object.entries(scores).reduce((a, b) => (b[1] > a[1] ? b : a))[0];
    setResult(best);
  }, [answers]);

  return (
    <div className="max-w-md mx-auto p-4 space-y-4 text-center">
      <h2 className="text-2xl font-semibold">You are most like a {result}!</h2>
      <img
        src={animalImages[result]}
        alt={result}
        width={256}
        height={256}
        className="mx-auto"
      />
      <Share text={`I am most like a ${result}! ${url}`} />
      <button
        className="px-4 py-2 bg-secondary text-secondary-foreground rounded hover:bg-secondary/80"
        onClick={onRetake}
      >
        Retake Quiz
      </button>
    </div>
  );
}
